"""Aplicación REST para la descarga de imágenes de distintos servicios

Esta aplicación permite descargar imágenes por satélite de servicios
como Google Maps o Bing Maps. Los módulos implementados son independientes
y es posible añadir más de forma transparente.

Example:
    Para ejecutar este script se debe ejecutar en una terminal en la carpeta
    donde se almacene este archivo el siguiente código:

            $ python ApiRest.py

    Tras esto se podrán realizar llamadas a la API desde el servidor:

            http://localhost:5000/

Attributes:
    app(Flask): Instancia flask para dotar de servicios REST a la aplicación.
    api(Api): Instancia de Api para añadir rutas de forma más simple a
        la aplicación.
    servicios(list[str]): Lista de servicios que están disponibles en la API.
        Se leen del fichero conf.json en la misma carpeta que este fichero.
    ruta_modulos(list[str]): Ruta a cada uno de los módulos. Se leen del fichero
        conf.json
    modulos(list[module]): Módulos importados para utilizarlos en este fichero.
    callable_services(list[servicio]): Lista de instancias de las clases de cada
        uno de los servicios disponibles. Utilizado para hacer las llamadas a las
        distintas funciones de los módulos.

"""

from flask import Flask
from flask import make_response
from flask import request
from flask import send_file
import io
from flask_restful import Api, Resource
import json
import importlib
import inspect
from ApiExceptions import *
import geojson
import os

app = None
api = None
conf = None
servicios = []
ruta_modulos = []
modulos = []
callable_services = []



[docs]def initialize():
    """Función para inicializar los atributos de la aplicación

        Inicializa los parámetros del fichero para acceder a estos
        y poner en funcionamiento la API REST. Estos son: conf, servicios,
        ruta_modulos, modulos, api, app, callable_services

        """
    global conf, servicios, ruta_modulos, modulos, api, app, callable_services

    #Abrimos y guardamos el fichero de configuración
    current_directory = os.path.dirname(__file__)
    with open(os.path.join(current_directory, "conf.json"), "r") as conf_file:
        conf = dict(json.loads(conf_file.read()))

    #Obtenemos los nombres de los servicios(lista servicios),
    #la ruta de los modulos(lista ruta_modulos)
    #y los cargamos en memoria(lista modulos)
    for servicio, archivo in conf['servicios'].items():
        servicios.append(servicio.lower())
        ruta_modulos.append(archivo.replace("/", "."))

    for modulo in ruta_modulos:
        modulos.append(importlib.import_module(modulo))

    #Instanciamos las clases y las guardamos en callable_services
    for x in range(len(modulos)):
        class_names = inspect.getmembers(modulos[x], inspect.isclass)
        class_name = [serv[1] for serv in class_names if serv[0].lower() == servicios[x]][0]
        callable_services.append(class_name(conf['keys'][servicios[x]]))

    #Creamos la Api Rest
    app = Flask(__name__)
    api = Api(app)

    #Añadimos las rutas
    api.add_resource(Image, '/getImage')
    api.add_resource(Metadata, '/getMetadata')
    api.add_resource(Servicios, '/getServices')
    api.add_resource(ImagePreview, '/getPreview')

                    - servicio
                        - Descripción: Servicio del que se quiere obtener la imagen.

                    - id
                        - Descripción: Id de la imagen
                    - point
                        - Descripción: Punto perteneciente a la imagen en latitud y longitud.
                        - Formato: "latitud,longitud"
                        - Ejemplo: "41.800507792693544,-2.304528141507549"
                - Opcionales:
                    - size
                        - Descripción: Tamaño en píxeles de la imagen.
                        - Formato: "altura,anchura"
                        - Ejemplo: "1000,1000"
                    - format
                        - Descripción: Formato de la imagen.
                    - resolution
                        - Descripción: Resolución máxima en metros/pixel de la imagen que queremos obtener.
                        - Ejemplo: "3.5"
                    - beforeDate
                        - Descripción: Obtener únicamente imágenes anteriores a la fecha dada.
                        - Formato: "dd-mm-aaaa"
                        - Ejemplo: "12-09-2005"
                    - afterDate
                        - Descripción: Obtener únicamente imágenes posteriores a la fecha dada.
                        - Formato: "dd-mm-aaaa"
                        - Ejemplo: "12-09-2005"
                    - maxResults
                        - Descripción: Número máximo de resultados que queremos obtener por servicio
                        - Ejemplo: "5"
                    - itemTypes
                        - Descripción: Tipos de imágenes que se quiere que se devuelvan
                        - Ejemplo: "PSScene4Band,PSOrthoTile"

        Returns:
            Response: Respuesta con la imagen solicitada en bytes con los metadatos
            en la cabecera y el código HTTP 200.

        Returns:
            Response: Respuesta con el estado de una imagen que se ha solicitado descargar
            pero necesita tiempo su activación y el código HTTP 202.

        Returns:
            Response: Respuesta con un mensaje de error si alguno ha ocurrido con
            el código HTTP 400.

    """


[docs]    def get(self):

        global callable_services
        global servicios

        #Añadimos los parámetros por defecto a los parámetros del usuario
        user_param = dict(request.args)
        final_parameters = conf['defaultParameters'].copy()
        final_parameters.update(user_param)

        #Obtenemos el indice del servicio solicitado
        try:
            idx = servicios.index(final_parameters['servicio'].lower())
        except ValueError:
            return make_response("El servicio '{}' indicado no existe.".format(final_parameters['servicio']), 400)
        except KeyError:
            return make_response("Se debe indicar el servicio del que se quiere obtener la imagen", 400)

        #Hacemos la petición con la función getImage
        try:
            resp, metadata = callable_services[idx].getImage(final_parameters)
        except ActivatingException as ae:
            return make_response(ae.args[0], 202)
        except ApiException as ae:
            return make_response(ae.args[0], 400)


        #Creamos la respuesta con la imagen y le añadimos los metadatos
        response = make_response(send_file(io.BytesIO(resp),
                         mimetype='application/octet-stream'))

        response.headers['metadata'] = geojson.dumps(metadata)

        return response

[docs]    def get(self):

        global callable_services
        global servicios

        #Añadimos los parámetros por defecto a los parámetros del usuario
        user_param = dict(request.args)
        final_parameters = conf['defaultParameters'].copy()
        final_parameters.update(user_param)

        #Si no se especifican servicios se devuelven todos
        if not 'servicio' in final_parameters:
            final_parameters['servicio'] = servicios
        else:
            final_parameters['servicio'] = final_parameters['servicio'].lower().split(',')

        #Creamos una lista y almacenamos los metadatos de cada servicio
        metadata_dict = []
        for servicio in final_parameters['servicio']:
            try:
                if not servicio in servicios:
                    return make_response("El servicio '{}' no existe en el sistema".format(servicio), 400)
                idx = servicios.index(servicio)
                resp = callable_services[idx].getMetadata(final_parameters)
                metadata_dict += resp
            except ApiException as e:
                return make_response(e.args[0], 400)

        #Creamos el GeoJson con los metadatos de los servicios
        geojson_dict = {
            "type": "FeatureCollection",
            "features": [metadata_dict]
        }
        geojson_str = geojson.dumps(geojson_dict, indent=2)
        return make_response(geojson_str, 200)
