[docs]    def dataRequest(self, user_params):
        """Función para obtener el link de una imagen y sus metadatos

        Dados unos parámetros de entrada devuelve el link de la imagen
        y sus metadatos en formato GeoJSON.

        Parameters:
            user_params (dict[str->str]):
                Diccionario con los valores de entrada del usuario.

                - point
                    - Descripción: Punto perteneciente a la imagen en latitud y longitud.
                    - Formato: "latitud,longitud"
                    - Ejemplo: "41.800507792693544,-2.304528141507549"
                - size
                    - Descripción: Tamaño en píxeles de la imagen.
                    - Formato: "altura,anchura"
                    - Ejemplo: "1000,1000"
                - format
                    - Descripción: Formato de la imagen.
                - resolution
                    - Descripción: Resolución máxima en metros/pixel de la imagen que queremos obtener.
                    - Ejemplo: "3.5"
                - beforeDate
                    - Descripción: Obtener únicamente imágenes anteriores a la fecha dada.
                    - Formato: "dd-mm-aaaa"
                    - Ejemplo: "12-09-2005"
                - afterDate
                    - Descripción: Obtener únicamente imágenes posteriores a la fecha dada.
                    - Formato: "dd-mm-aaaa"
                    - Ejemplo: "12-09-2005"

        Returns:
            Tuple(str, dict[str->any]): Tupla con el link de la imagen y sus metadatos formato GeoJSON

        Raises:
            ParametersException: Si se han introducido erróneamente valores de entrada
            ApiException: Si la consulta a Bing Maps ha dado error
            NotFoundImage: Si no hay imagen disponible para los parámetros dados.

        """

        #Comprobamos que no falte ningún parámetro obligatorio
        for param in self.required_params:
            if not param in user_params:
                raise ParametersException("Error en Bing Maps: Se debe introducir el parámetro '{}' en la consulta o establecer"
                                "un valor predeterminado en el fichero de configuración.".format(param))

        #Convertimos el tamaño de str a int
        try:
            size = user_params['size'].split(',')
            size = [int(size[0]), int(size[1])]
        except ValueError:
            raise ApiException("Parámetro 'Size' introducido erróneamente. El formato es: 'altura,anchura'")

        #Obtenemos la ruta del servidor y establecemos el zoom maximo
        server_route = 'https://dev.virtualearth.net/REST/v1/Imagery/Metadata/Aerial'
        current_zoom = user_params['zoom']

        metadatos = None
        request_link = None

        #Bucle para pedir una imagen de Bing Maps. Si de un zoom no tuviera, pediría del anterior hasta
        #que encontrara una imagen o la resolución fuera mayor que la pedida.
        while metadatos is None and self.zoom_dict[str(current_zoom)] <= float(user_params['resolution']) and current_zoom >= 1:

            request_link = server_route + \
                         '/{}'.format(user_params['point']) + \
                         '/{}'.format(current_zoom)+ \
                         '?mapSize={},{}'.format(size[0], size[1]) + \
                         '&key={}'.format(self.key)

            response = requests.get(request_link)

            if response.status_code != 200:
                raise ApiException("Error en Bing Maps: {}".format(response.text))

            res = json.loads(response.text)
            try:
                resources = res['resourceSets'][0]['resources'][0]
            except:
                raise NotFoundImage("Error en Bing Maps: No hay imagen de este tipo para esta localizacion.")
            start_date = resources['vintageStart']
            if start_date is not None:
                metadatos = res
            else:
                current_zoom-=1
        if metadatos is None:
            raise NotFoundImage("Error en Bing Maps: No hay imagen de este tipo para esta localizacion.")


        #Convertimos las fechas de str a date
        startDate = resources['vintageStart'].split()
        startDate = date(int(startDate[2]), self.month_dict[startDate[1]], int(startDate[0]))

        endDate = resources['vintageEnd'].split()
        endDate = date(int(endDate[2]), self.month_dict[endDate[1]], int(endDate[0]))

        #Comprobamos si las fechas entran dentro del rango solicitado
        beforeDate = [int(x) for x in user_params['beforeDate'].split("-")]
        beforeDate = date(beforeDate[2], beforeDate[1], beforeDate[0])
        if beforeDate < endDate:
            raise NotFoundImage("Error en Bing Maps: Fecha no disponible para esta imagen")

        afterDate = [int(x) for x in user_params['afterDate'].split("-")]
        afterDate = date(afterDate[2], afterDate[1], afterDate[0])
        if afterDate > startDate:
            raise NotFoundImage("Error en Bing Maps: Fecha no disponible para esta imagen")

        #Calculamos las coordenadas
        coords = [float(x) for x in user_params['point'].split(",")]
        geometry = Utils.centerCoordsToGeojsonGeometry(coords, size, self.zoom_dict[str(current_zoom)])

        #Creamos el GeoJSON
        metadata = {
            "type": "Feature",
            "geometry": {
                "type": "Polygon",
                "coordinates": geometry
            },
            "properties": {
                "service": "Bing Maps",
                "id": None,
                "punto_central": user_params['point'],
                "acquired_date": '{}<->{}'.format(startDate, endDate),
                "download_date": date.today().strftime("%d-%m-%G"),
                "size": "{},{}".format(size[0], size[1]),
                "resolution": self.zoom_dict[str(current_zoom)]
            }
        }

        #Link de la imagen
        img_link = request_link.replace("Metadata","Map")

        return img_link, metadata




[docs]    def getImage(self, user_params):
        """Función para obtener una imagen y sus metadatos

        Dados unos parámetros de entrada devuelve una imagen
        en bytes y sus metadatos en formato GeoJSON.

        Args:
            user_params (dict[str->str]): Diccionario con los valores de entrada
                del usuario. Mismos parámetros que la función :py:meth:`BingMaps.BingMaps.dataRequest`

        Returns:
            Tuple(bytes, Dict[str->any]): Tupla con la imagen en bytes y sus metadatos formato GeoJSON

        """

        img_link, metadata = self.dataRequest(user_params)

        resp = requests.get(img_link)

        #Se ha producido un error
        if resp.status_code != 200:
            raise ApiException(resp.text)
        else:
            return resp.content, metadata
