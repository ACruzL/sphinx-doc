    Attributes:
        all_item_types (list[str]): Todos los tipos de satélite de Planet Labs.
            Parámetro necesario para hacer consultas
        key (str): Clave para acceder a la Api de Planet Labs

    def __init__(self, key):

        self.all_item_types = ["PSScene3Band",
                          "PSScene4Band",
                          "PSOrthoTile",
                          "REOrthoTile",
                          "REScene",
                          "SkySatScene",
                          "SkySatCollect",
                          "SkySatVideo",
                          "Landsat8L1G",
                          "Sentinel2L1C"]
        self.all_item_types = ["Sentinel2L1C"]
        self.required_params = ["point", "resolution", "beforeDate", "afterDate", "maxResults"]
        self.key = key

        self.client = api.ClientV1(api_key=self.key)


[docs]    def dataRequest(self, user_params):
        """Función para obtener los metadatos de las imágenes coincidentes con los parámetros de entrada.

        Dados unos parámetros de entrada devuelve los metadatos que
        Planet Labs ofrece de las imágenes que coincidan con estos.

        Parameters:
            user_params (dict[str->str]):
                Diccionario con los valores de entrada del usuario.

                - point
                    - Descripción: Punto perteneciente a la imagen en latitud y longitud.
                    - Formato: "latitud,longitud"
                    - Ejemplo: "41.800507792693544,-2.304528141507549"
                - resolution
                    - Descripción: Resolución máxima en metros/pixel de la imagen que queremos obtener.
                    - Ejemplo: "3.5"
                - beforeDate
                    - Descripción: Obtener únicamente imágenes anteriores a la fecha dada.
                    - Formato: "dd-mm-aaaa"
                    - Ejemplo: "12-09-2005"
                - afterDate
                    - Descripción: Obtener únicamente imágenes posteriores a la fecha dada.
                    - Formato: "dd-mm-aaaa"
                    - Ejemplo: "12-09-2005"
                - maxResults
                    - Descripción: Número máximo de resultados que queremos obtener por servicio
                    - Ejemplo: "5"
                - itemTypes
                    - Descripción: Tipos de imágenes que se quiere que se devuelvan
                    - Ejemplo: "PSScene4Band,PSOrthoTile"


        Returns:
            dict[str->any]: Diccionario con los metadatos de las imágenes devueltas

        Raises:
            ParametersException: Si se han introducido erróneamente valores de entrada.

        """

        #Comprobamos que no falte ningún parámetro obligatorio
        for param in self.required_params:
            if not param in user_params:
                raise ParametersException("Se debe introducir el parámetro '{}' en la consulta o establecer"
                                "un valor predeterminado en el fichero de configuración.".format(param))

        #Creamos el filtro para hacer la búsqueda
        filter = self.create_filter(user_params)

        #Si no se han especificado tipos, se buscarán todos
        if 'item_types' not in user_params:
            user_params['item_types'] = self.all_item_types
        else:
            user_params['item_types'] = user_params['item_types'].replace(" ","").split(',')

        request = api.filters.build_search_request(filter, user_params['item_types'])
        results = self.client.quick_search(request)
        
        return results._json_stream(int(user_params['maxResults']))



[docs]    def create_filter(self, user_params):
        """Función para crear el filtro de Planet Labs

        Dados unos parámetros de entrada, devuelve los filtros
        utilizados por Planet Labs para poder hacer peticiones
        a su servicio REST.

        Args:
            user_params (dict[str->any]): Diccionario con los valores de entrada
                del usuario. Mismos parámetros que la función :py:meth:`PlanetLabs.PlanetLabs.dataRequest`

        Returns:
            list(filter): Lista de filtros

        """

        filters = []

        #Filtro para que la imagen tenga permisos de descarga
        filters.append(api.filters.permission_filter('assets.visual:download'))

        #Filtro para que las fechas estén dentro del rango solicitado
        if 'beforeDate' in user_params:
            date_list = [int(x) for x in user_params['beforeDate'].split("-")]
            asked_date = datetime(date_list[2], date_list[1], date_list[0])
            filters.append(api.filters.date_range("acquired", lte= asked_date))
        if 'afterDate' in user_params:
            date_list = [int(x) for x in user_params['afterDate'].split("-")]
            asked_date = datetime(date_list[2], date_list[1], date_list[0])
            filters.append(api.filters.date_range("acquired", gte=asked_date))

        #Filtro para que el punto esté pertenezca a las imágenes
        if 'point' in user_params:
            point = [float(x) for x in user_params['point'].split(",")]
            geoJson = geojson.Point(point)
            filters.append(api.filters.geom_filter(geoJson))

        #Filtro para que la resolución sea menor que un valor dado
        if 'resolution' in user_params:
            filters.append(api.filters.range_filter("gsd", lte=float(user_params['resolution'])))

        final_filter = api.filters.and_filter(*filters)

        return final_filter



[docs]    def get_asset(self, id):
        """Función para obtener los assets de una imagen dada por su id.

        Dado un id, devuelve los links y metadatos de la imagen
        que corresponda a ese mismo id.

        Args:
            id (str): Id de la imagen

        Returns:
            dict(any->any): Assets de la imagen

        """

        filter = api.filters.string_filter('id', id)

        request = api.filters.build_search_request(filter, self.all_item_types)

        results = self.client.quick_search(request)
        #Si no se ha devuelto nada de la petición es porque el id es incorrecto
        if len(list(results.items_iter(1))) == 0:
            raise ApiException("El id insertado es incorrecto.")

        asset = self.client.get_assets(next(results.items_iter(1))).get()

        return asset

[docs]    def getDownloadLink(self, id):
        """Función que devuelve el link de descarga de una imagen

        Dado un id, devuelve el link de descarga de esa imagen. Es
        necesario que previamente se active la misma para poder acceder
        a la descarga.

        Args:
            id (str): Id de la imagen

        Returns:
            str: Link de descarga de la imagen o un mensaje con el estado
            de la imagen si no estuviera activada.

        Raises:
            NotActiveException: Si el recurso se está activando o no está activo

        """

        asset = self.get_asset(id)['visual']
        if not 'location' in asset:
            if asset['status'] == 'activating':
                raise NotActiveException("Activating")
            elif asset['status'] == 'inactive':
                raise NotActiveException("Inactive")
        else:
            return asset['location']



[docs]    def getMetadata(self, user_params):
        """Función para obtener los metadatos de ciertas imágenes

        Dados unos parámetros de entrada devuelve los metadatos
        de las imagenes en formato GeoJSON. La diferencia con los
        metadatos que ofrece Planet Labs es que estos están
        seleccionados para una mejor visualización.

        Args:
            user_params (dict[str->any]): Diccionario con los valores de entrada
                del usuario. Mismos parámetros que la función :py:meth:`PlanetLabs.PlanetLabs.dataRequest`

        Returns:
            list(dict[str->any]): Lista de metadatos formato GeoJSON

        Raises:
            ApiException: Si se produce un error en la API de Planet Labs

        """

        try:
            metadata = self.dataRequest(user_params)
        except api.exceptions.APIException as ae:
            raise ApiException("Error en Planet Labs: " + ae.args[0])


        feature_list = []
        for image in metadata['features']:
            data = {
                "type": "Feature",
                "geometry": image['geometry'],
                "properties": {
                    "service": "Planet Labs",
                    "id": image['id'],
                    "punto_central": None,
                    "acquired_date": arrow.get(image['properties']['acquired']).datetime.strftime("%d-%m-%G"),
                    "download_date": date.today().strftime("%d-%m-%G"),
                    "size": '{},{}'.format(image['properties']['rows'],image['properties']['columns']),
                    "resolution": image['properties']['gsd']
                }
            }
            feature_list.append(data)


        return feature_list

        """Función para obtener los metadatos de ciertas imágenes

        Dados un id de una imagen, devuelve los metadatos
        de esta misma en formato GeoJSON.

        Args:
            id (str): Id de la imagen

        Returns:
            dict[str->any]: Lista de metadatos formato GeoJSON

        Returns:
            str: Mensaje de error si se produce una excepción

        Raises:
            ApiException: Si se produce un error en la API

        """

        try:
            id_filter = api.filters.string_filter('id', id)
            request = api.filters.build_search_request(id_filter, self.all_item_types)
            results = self.client.quick_search(request)
            metadata = results._json_stream(1)
        except api.exceptions.APIException as ae:
            raise ApiException("Error en Planet Labs: " + ae.args[0])

        for image in metadata['features']:
            final_metadata = {
                "type": "Feature",
                "geometry": image['geometry'],
                "properties": {
                    "service": "Planet Labs",
                    "id": image['id'],
                    "punto_central": None,
                    "acquired_date": arrow.get(image['properties']['acquired']).datetime.strftime("%d-%m-%G"),
                    "download_date": date.today().strftime("%d-%m-%G"),
                    "size": '{},{}'.format(image['properties']['rows'],image['properties']['columns']),
                    "resolution": image['properties']['gsd']
                }
            }

        return final_metadata



[docs]    def getImage(self, user_params):
        """Función para obtener una imagen y sus metadatos

        Dado un diccionario con el parámetro id correspondiente
        al id de la imagen, devuelve la imagenen bytes y sus
        metadatos en formato GeoJson.

        Args:
            user_params (dict[str->any]): Diccionario con el
                parámetro id.

        Returns:
            Tuple(bytes, Dict[str->any]): Tupla con la imagen en bytes y sus metadatos formato GeoJSON

        Raises:
            ApiException: Si se produce un error en la API
            ActivatingException: Si el recurso no está aun activo

        """
        try:
            img_link = self.getDownloadLink(user_params['id'])
        except NotActiveException as ae:
            if ae.args[0] == 'Inactive':
                self.activate(user_params['id'])
                raise ActivatingException("Se ha mandado la peticion de activar el recurso. Pruebe en unos momentos.")
            elif ae.args[0] == 'Activating':
                raise ActivatingException("El recurso se está activando. Pruebe en unos momentos.")
        except api.exceptions.APIException as ae:
            raise ApiException(ae.args[0])
        except KeyError:
            raise ApiException("Se debe introducir el id de la imagen.")


        try:
            resp = requests.get(img_link)
            metadata = self.getMetadataById(user_params['id'])
        except api.exceptions.APIException as ae:
            raise ApiException(ae.args[0])

        return resp.content, metadata



[docs]    def getPreview(self, user_params):
        """Función para obtener la miniatura de una imagen

        Dado el id de la imagen y una anchura se devuelve
        la imagen en formato png.

        Args:
            user_params (dict[str->str])
                Diccionario con los parámetros:
                    - Obligatorio:
                        - id
                            - Descripción: Id de la imagen
                    - Opcional
                        - size
                            - Descripción: Tamaño de la imagen
                            - Ejemplo: "2048,2048"

        Returns:
            Bytes: Imagen en bytes

        Raises:
            ApiException: Si se produce un error en la API
            ParametersException: Si la anchura no entra dentro del rango posible.

        """

        try:
            filter = api.filters.string_filter('id', user_params['id'])
            request = api.filters.build_search_request(filter, self.all_item_types)
            results = self.client.quick_search(request)
            #Si no se ha devuelto nada de la petición es porque el id es incorrecto
            if len(list(results.items_iter(1))) == 0:
                raise ApiException("El id insertado es incorrecto.")
            thumb_link = next(results.items_iter(1))['_links']['thumbnail']
        except api.exceptions.APIException as ae:
            raise ApiException(ae.args[0])

        width = int(user_params['size'].split(",")[0])
        query_dict = {'width': width}
        resp = requests.get(
                thumb_link, query_dict,
                auth=HTTPBasicAuth(self.key, '')
            )

        if resp.status_code != 200:
            raise ParametersException(resp.text)
        else:
            return resp.content
