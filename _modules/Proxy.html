"""Proxy para acceder a la API REST de Satellite Images.

Este módulo tiene funciones específicas para obtener imágenes
por satélite de los distintos servicios implementados por
la API REST Satellite Images de forma homogénea.

"""

import requests
import json
import os

#Cargamos los parámetros por defecto
current_directory = os.path.dirname(__file__)
with open(os.path.join(current_directory, "conf.json"), "r") as conf_file:
    conf = dict(json.loads(conf_file.read()))
    default_params = conf['defaultParameters']
    server_route = conf['server']

[docs]def getMetadata(point, **kwargs):
    """Función para obtener metadatos de imágenes que coincidan con los parámetros

        Esta clase llama a la ruta **/getMetadadata** paraobtener los
        metadatos de la imágenes coincidentes con los parámetros dados
        en la consulta.

        Parameters:
            point (str)

                - Descripción: Punto perteneciente a la imagen en latitud y longitud.
                - Formato: "latitud,longitud"
                - Ejemplo: "41.800507792693544,-2.304528141507549"

            **kwargs (dict[str->str])
                Lista de parámetros que haya introducido el usuario. Estos pueden incluir:

                - servicio
                    - Descripción: Servicio  o servicios de los que se quieren obtener metadatos.
                - size
                    - Descripción: Tamaño en píxeles de la imagen.
                    - Formato: "altura,anchura"
                    - Ejemplo: "1000,1000"
                - format
                    - Descripción: Formato de la imagen.
                - resolution
                    - Descripción: Resolución máxima en metros/pixel de la imagen que queremos obtener.
                    - Ejemplo: "3.5"
                - beforeDate
                    - Descripción: Obtener únicamente imágenes anteriores a la fecha dada.
                    - Formato: "dd-mm-aaaa"
                    - Ejemplo: "12-09-2005"
                - afterDate
                    - Descripción: Obtener únicamente imágenes posteriores a la fecha dada.
                    - Formato: "dd-mm-aaaa"
                    - Ejemplo: "12-09-2005"
                - maxResults
                    - Descripción: Resolución máxima en metros/pixel de la imagen que queremos obtener.
                    - Ejemplo: "3.5"
                - itemTypes
                    - Descripción: Tipos de imágenes que se quiere que se devuelvan (Para Planet Labs)
                    - Ejemplo: "PSScene4Band,PSOrthoTile"

        Returns:
            str: Metadatos en formato GeoJSON o un mensaje de error si se hubiera
            producido alguno.

    """

    request_link = server_route + "getMetadata?point={}".format(point)
    for attribute, value in kwargs.items():
        request_link += "&{}={}".format(attribute, value)

    response = requests.get(request_link)

    return response.text




[docs]def getImage(provider, **kwargs):
    """Función para obtener una imagen

        Dados unos parámetros, se devuelve la imagen coincidente
        con estos además de sus metadatos correspondientes. En
        algunos servicios el parámetro *point* será obligatorio
        mientras que en otros lo séra el parámetro *id*, dependiendo
        de las características de este servicio.

        Parameters:
            provider (str): Servicio del que se quiere obtener la imagen

                - Descripción: Proveedor o servicio del que se quiere obtener la imagen

            **kwargs (dict[str->str])
                Lista de parámetros que haya introducido el usuario. Estos pueden incluir:

                - point
                    - Descripción: Punto perteneciente a la imagen en latitud y longitud.
                    - Formato: "latitud,longitud"
                    - Ejemplo: "41.800507792693544,-2.304528141507549"
                - size
                    - Descripción: Tamaño en píxeles de la imagen.
                    - Formato: "altura,anchura"
                    - Ejemplo: "1000,1000"
                - id
                    - Descripción: Id de la imagen que se quiere obtener

        Returns:
            Tuple(bytes, str): Imagen en bytes y sus metadatos

        Returns:
            Tuple(str, None): Mensaje de error si se ha producido alguno cuando
            se intentaba obtener la imagen

    """

    request_link = server_route + "getImage?servicio={}".format(provider)
    for attribute, value in kwargs.items():
        request_link += "&{}={}".format(attribute, value)

    response = requests.get(request_link)

    if response.status_code == 200:
        return response.content, response.headers['metadata']
    else:
        return response.text, None
