[docs]    def test_getImage_googlemaps(self):
        """Test para la función getImage con el servicio Google Maps

        """
        # Debería devolver la imagen
        img1, metadata1 = Proxy.getImage("googlemaps", point="40.714728,-73.9986729")
        self.assertIsInstance(img1, bytes)
        self.assertIsInstance(metadata1, str)

        #Debería devolver el mensaje de error y null porque falta el parámetro "point"
        img1, metadata1 = Proxy.getImage("googlemaps")
        self.assertIsInstance(img1, str)
        self.assertIsNone(metadata1)

        #Debería devolver la imagen
        img1, metadata1 = Proxy.getImage("googlemaps", point="40.714728,-73.9986729", size="1000,1000", format='jpg')
        self.assertIsInstance(img1, bytes)
        self.assertIsInstance(metadata1, str)

        #Debería devolver el mensaje de error y null porque el formato de "size" es "altura,anchura" no "alturaxanchura"
        img1, metadata1 = Proxy.getImage("googlemaps", point="40.714728,-73.9986729", size="1000x1000")
        self.assertIsInstance(img1, str)
        self.assertIsNone(metadata1)

        #Debería devolver el mensaje de error y null
        img1, metadata1 = Proxy.getImage("googlemaps", point="40.aaaaa,-73.9986729,-2.304528141507549")
        self.assertIsInstance(img1, str)
        self.assertIsNone(metadata1)

[docs]    def test_getMetadata(self):
        """Test para la función getMetadata

        """
        # Debería devolver el mensaje de error
        metadata = Proxy.getMetadata("40.714728,-73.9986729", servicio="bingmaps,servicioinventado")
        with self.assertRaises(json.decoder.JSONDecodeError):
            json.loads(metadata)

        # Debería devover los metadatos
        metadata = Proxy.getMetadata("40.714728,-73.9986729", servicio="bingmaps,googlemaps,planetlabs")
        json.loads(metadata)

        #Debería devolver el mensaje de error porque falta el parámetro "point"
        metadata = Proxy.getMetadata("bingmaps,googlemaps,planetlabs")
        with self.assertRaises(json.decoder.JSONDecodeError):
            json.loads(metadata)

        # Debería devover los metadatos
        metadata = Proxy.getMetadata("40.714728,-73.9986729", servicio="bingmaps,googlemaps,planetlabs", size="1000,1000", format='jpg')
        json.loads(metadata)

        #Debería devolver el mensaje de error y null porque el formato de "size" es "altura,anchura" no "alturaxanchura"
        metadata = Proxy.getMetadata("40.714728,-73.9986729", servicio="bingmaps,googlemaps,planetlabs", size="1000x1000")
        with self.assertRaises(json.decoder.JSONDecodeError):
            json.loads(metadata)

        #Debería devolver el mensaje de error y null
        metadata = Proxy.getMetadata("-73.9aaaaa986729,-2.304528141507549")
        with self.assertRaises(json.decoder.JSONDecodeError):
            json.loads(metadata)
