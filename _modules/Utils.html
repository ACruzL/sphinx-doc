[docs]def metersToLat(meters):
    """Convierte metros a diferencia de latitud

    Parameters:
        meters (float): Metros

    Returns:
        float: Diferencia de latitud

    """

    metersPerLat = 111000
    return meters/metersPerLat



[docs]def metersToLon(meters, lat):
    """Convierte metros a diferencia de longitud

    Parameters:
        meters (float): Metros
        lat (float): Latitud que corresponde a la longitud que queremos obtener

    Returns:
        float: Diferencia de longitud

    """
    metersPerLon = 111000 * math.cos(lat)
    return meters/metersPerLon



[docs]def centerCoordsToGeojsonGeometry(coords, size, resolution):
    """Obtiene las coordenadas de las esquinas de una imagen

    Parameters:
        coords(list[latitud, longitud]): Latitud y longitud del punto central de la imagen
        size(list[altura,anchura]): Altura y anchura de la imagen en píxeles
        resolution (float): Resolution en metros/pixel

    Returns:
        list[list[latitud,longitud]]]: Geometría del polígono en formato GeoJSON

    """
    difLat = metersToLat(resolution * size[0] / 2)
    difLon = metersToLon(resolution * size[1] / 2, size[0])

    geometry = [[   [coords[0]-difLat, coords[1]-difLon], [coords[0]+difLat, coords[1]-difLon],
                    [coords[0]+difLat, coords[1]+difLon], [coords[0]-difLat, coords[1]+difLon],
                    [coords[0]-difLat, coords[1]-difLon] ]]

    return geometry
